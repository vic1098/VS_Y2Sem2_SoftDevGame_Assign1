#include <iostream>

using namespace std;


void gcd(int a, int b)
{
	int c = a + b;

	c = c / 2;

	cout << "A = " << a << ", B = " << b << ". The GCD = " << c << endl;
}

int main()
{
	int num1, num2;

	cout << "Please enter a value for num1: ";
	cin >> num1;
	cout << endl;

	cout << "Please enter a value for num2: ";
	cin >> num2;
	cout << endl;

	gcd(num1, num2);
}
